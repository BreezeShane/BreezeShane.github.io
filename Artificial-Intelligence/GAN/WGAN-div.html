def gradient_penalty(D, x_r, x_f, p):
    t = torch.rand(batchsz, 1).cuda()
    t = t.expand_as(x_r)

    x_hat = (1 - t) * x_r + t * x_f
    x_hat.requires_grad_(True)

    pred = D(x_hat)
    grads = autograd.grad(outputs=pred,
                          inputs=x_hat,
                          grad_outputs=torch.ones_like(pred),
                          create_graph=True,
                          retain_graph=True,
                          only_inputs=True)[0]
    gp = torch.pow(torch.square(grads), p/2)
    return gp.mean()
